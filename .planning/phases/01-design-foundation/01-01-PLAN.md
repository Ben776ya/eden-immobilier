```typescript
import { Outfit } from 'next/font/google'

const outfit = Outfit({

  variable: '--font-outfit',
  display: 'swap',
  weight: ['300', '400', '500', '600'],
})

export const metadata: Metadata = {
  title: 'Eden Immobilier — Proprietes de Prestige',
  description:
    "Decouvrez notre selection exclusive de proprietes de luxe sur la Cote d'Azur, a Paris et dans les plus beaux domaines de France.",
}

export default function RootLayout({
  children,
}: {
  children: React.ReactNode
}) {
  return (
    <html lang="fr" className={outfit.variable}>
      <body className="bg-eden-surface text-eden-text font-sans antialiased">
        <Navbar />
        <main>{children}</main>
        <Footer />
      </body>
    </html>
  )
}
```

CRITICAL details:
- The CSS variable MUST be `--font-outfit` — this exact string must match the `var(--font-outfit)` in tailwind.config.ts fontFamily.sans. If these don't match, Outfit silently fails to load and the browser falls back to system-ui (Pitfall 3 from RESEARCH.md).
- Weight array is `['300', '400', '500', '600']` per TYP-03. NOT `['300', '400', '500', '600', '700']` — the requirement specifies 4 weights.
- The body className uses new tokens: `bg-eden-surface` (was bg-eden-bg), `text-eden-text` (was text-eden-cream).
- The html element has only `outfit.variable` (was `${playfair.variable} ${dmSans.variable}`).
- Playfair_Display and DM_Sans imports are completely removed.
- Metadata title and description are preserved from the original (keep the French text with proper accents — use the exact strings from the existing file).
